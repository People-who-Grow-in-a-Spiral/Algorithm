j_money = s_money = int(input())

    if j_money >= cost[i]:
       j_stock += j_money // cost[i]
       j_money %= cost[i]
    if i > 2:
        if cost[i - 3] > cost[i - 2] > cost[i - 1]:

            s_money %= cost[i]
        elif cost[i - 3] < cost[i - 2] < cost[i - 1]:
            s_money += s_stock * cost[i]
            s_stock = 0
            
res_j = j_money + cost[-1] * j_stock
res_s = s_money + cost[-1] * s_stock

if res_j > res_s:
    print("BNP")
elif res_j < res_s:
    print("TIMING")
else:
    print("SAMESAME")
