```python
# BOJ
# S5 - 20546(🐜 기적의 매매법 🐜)

money = [int(input())] * 2
cost = list(map(int, input().split()))
stocks = [0] * 2

def buy(money, cost, stocks):
    stock = money // cost
    money -= stock * cost
    stocks += stock
    
    return (money, stocks)

def sell(money, cost, stocks):
    money += cost * stocks
    
    return (money, stocks)

def check_sungmin(idx, money):
    up = down = 0
    for i in range(idx - 2, idx):
        if cost[i] > cost[i - 1]:
            up += 2
        elif cost[i] < cost[i - 1]:
            down += 2
                
    if up == 2:
        return 1
    elif down == 2:
        return 0
    else:
        return -1

stock = money[0] // cost[0]
for i in range(2):
    money[i] -= stock * cost[0]
    stocks[i] += stock

for i in range(1, len(cost)):
    if money[0] >= cost[i]:
        j_m, j_s = buy(money[0], cost[i], stocks[0])
        money[0] = j_m
        stocks[0] = j_s
    if i >= 3:
        check = check_sungmin(i, money[1])
        if check == 1:
            money[1] += cost[i] * stocks[1]
            stocks[1] = 0
        elif check == 0:
            s_m, s_s = buy(money[1], cost[i], stocks[1])
            money[1] = s_m
            stocks[1] = s_s
            
res_j = money[0] + cost[-1] * stocks[0]
res_s = money[1] + cost[-1] * stocks[1]

if res_j > res_s:
    print("BNP")
elif res_j < res_s:
    print("TIMING")
else:
    print("SAMESAME")
```
